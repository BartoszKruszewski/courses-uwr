def edge len angle collapse_speed
{
    if len > 10
    {
        edge len / collapse_speed, angle, collapse_speed
        left angle
        edge len / collapse_speed, angle, collapse_speed
        right angle * 2
        edge len / collapse_speed, angle, collapse_speed
        left angle
        edge len / collapse_speed, angle, collapse_speed
    }
    if len < 10 | len = 10
    {
        forward len
    }
}

def flake size angle collapse_speed
{
    repeat 3
    {
        edge size, angle
        right 120
    }
}

flake 200, 60, 2
